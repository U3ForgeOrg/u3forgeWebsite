function openWindow(title, description) {
  var link = document.createElement('a');
  link.href = 'data:text/html;charset=utf-8,' + encodeURI(`
    <html>
    <head>
      <title>${title}</title>
      <style>
        body {
          background-color: #888888;
          color: #000000;
          font-family: Arial, sans-serif;
          margin: 0;
        }
        h1 {
          text-align: center;
          color: #000000;
          margin: 20px;
        }
        p {
          text-align: center;
          margin: 20px;
        }
      </style>
    </head>
    <body>
      <h1>${title}</h1>
      <p>${description}</p>
    </body>
    </html>
  `);

  link.target = '_blank';
  link.click();
}

function openWindowWithVideo(title, description, videoId) {
  var link = document.createElement('a');
  link.href = 'data:text/html;charset=utf-8,' + encodeURI(`
    <html>
    <head>
      <title>${title}</title>
      <style>
        body {
          background-color: #888888;
          color: #000000;
          font-family: Arial, sans-serif;
          margin: 0;
        }
        h1 {
          text-align: center;
          color: #000000;
          margin: 20px;
        }
        p {
          text-align: center;
          margin: 20px;
        }
        .video-container {
          position: relative;
          padding-bottom: 56.25%;
          padding-top: 30px;
          height: 0;
          overflow: hidden;
        }
        .video-container iframe,
        .video-container object,
        .video-container embed {
          position: absolute;
          top: 0;
          left: 0;
          width: 100%;
          height: 100%;
        }
      </style>
    </head>
    <body>
      <h1>${title}</h1>
      <p>${description}</p>
      <div class="video-container">
        <iframe src="https://www.youtube.com/embed/${videoId}" frameborder="0" allowfullscreen></iframe>
      </div>
    </body>
    </html>
  `);

  link.target = '_blank';
  link.click();
}
